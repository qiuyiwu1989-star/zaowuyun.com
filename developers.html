01_// quickstart

3 分钟跑通第一个调用

step 01 · install

# npm
npm install @zaowuyun/sdk

# pip
pip install zaowuyun

step 02 · authenticate

import { Zaowuyun } from "@zaowuyun/sdk";

const client = new Zaowuyun({
  apiKey: process.env.ZAOWUYUN_API_KEY,
});

step 03 · call an agent

// 调用"商品摄影"Agent
const result = await client.agents.run({
  agent: "product_photo_v2",
  industry: "fmcg",
  inputs: {
    product_image: "https://...",
    platform: "tmall",
    festival: "double_11",
  },
  variants: 12,
});

console.log(result.images);
// → [{ url: "..." }, { url: "..." }, ...]
